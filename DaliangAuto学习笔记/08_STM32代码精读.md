### 超级循环解析
```c
static void Loop(void) {
    // 1. 轮速计算 (周期进行)
    WheelSpeed_Task();
    
    // 2. 屏幕通信管理
    Screen_Task();  // 心跳 + 超时检测
    
    // 3. 处理屏幕按键
    uint16_t addr, data;
    if (Screen_GetKeyAndAddr(&addr, &data)) {
        switch (addr) {
            case SCREEN_ADDR_MODE:
                SwitchMode(data);  // 切换工作模式
                break;
            case SCREEN_ADDR_GEAR:
                SwitchGear(data);
                break;
            case SCREEN_ADDR_SPEED:
                speed_scale = data;
                break;
        }
    }
    
    // 4. 跟随模式: 轮询Jetson指令
    if (g_steering_mode == MODE_FOLLOW) {
        FollowCmd_t cmd;
        if (TrainUART_PollFollowCmd(&cmd)) {
            s_follow_cmd = cmd;
            s_follow_cmd_valid = 1;
        }
    }
    
    // 5. 档位/刹车逻辑
    UpdateGearAndBrake();
    
    // 6. 物理档位开关消抖
    DebounceGearSwitches();
    
    // 7. CAN伺服心跳 (50ms周期)
    static uint32_t last_servo_hb = 0;
    if (HAL_GetTick() - last_servo_hb >= 50) {
        ServoCAN_SendHeartbeat();
        last_servo_hb = HAL_GetTick();
    }
    
    // 8. 处理CAN接收消息
    ServoCAN_ProcessRxMessages();
    
    // 9. 电机控制 (模式相关)
    UpdateMotorControl();
    
    // 10. 转向控制 (模式相关)
    UpdateSteeringControl();
    
    // 11. 训练数据上传 (50Hz)
    if (g_steering_mode == MODE_TRAINING) {
        static uint32_t last_train = 0;
        if (HAL_GetTick() - last_train >= 20) {
            SendTrainingData();
            last_train = HAL_GetTick();
        }
    }
}
```

## 二、RC输入捕获 (rc_input.c)

### PWM测量原理
```c
// RC遥控器输出PWM: 1000~2000us
// 中位: 1500us (对应转向/油门中立位置)

void RC_Input_OnCapture(TIM_HandleTypeDef* htim) {
    uint32_t ccr = htim->Instance->CCR1;
    uint32_t period;
    
    if (ccr > s_last_ccr) {
        period = ccr - s_last_ccr;
    } else {
        // 定时器溢出处理 (16位定时器)
        period = (0xFFFF - s_last_ccr) + ccr + 1;
    }
    s_last_ccr = ccr;
    
    // 更新脉冲宽度 (过滤异常值)
    if (period > 500 && period < 2500) {
        s_pulse_width = period;  // 单位: us
    }
}

// 映射到控制值
// 1000~1500us = 左转, 1500~2000us = 右转
// 1500~2000us = 油门加大
```

## 三、CAN伺服控制 (servo_can.c)

### 绝对位置控制
```c
void ServoCAN_GoToAbsolutePosition(int32_t position) {
    CanTxMsgTypeDef tx_msg;
    tx_msg.ExtId = 0;                    // 标准ID
    tx_msg.StdId = CAN_SERVO_ID;         // 0x64
    tx_msg.DLC = 8;                      // 8字节
    
    tx_msg.Data[0] = 0x22;               // 命令: 绝对位置
    tx_msg.Data[1] = 0x00;               // 子命令
    tx_msg.Data[2] = position & 0xFF;           // 位置 LSB
    tx_msg.Data[3] = (position >> 8) & 0xFF;
    tx_msg.Data[4] = (position >> 16) & 0xFF;
    tx_msg.Data[5] = (position >> 24) & 0xFF;   // 位置 MSB
    tx_msg.Data[6] = 0x00;               // 速度
    tx_msg.Data[7] = 0x00;               // 加速度
    
    HAL_CAN_AddTxMessage(&hcan, &tx_msg, 0);
}

// 转向角度 → 伺服位置映射
// STEERING_POSITION_RANGE: 伺服行程
// steer_offset: 中位偏移
void Steering_SetAngle(float angle_deg) {
    // angle_deg: -37° ~ +37°
    // 映射到伺服编码器位置
    float ratio = angle_deg / 37.0f;  // -1 ~ +1
    int32_t pos = STEERING_POSITION_MID + 
                  (int32_t)(ratio * STEERING_POSITION_RANGE / 2);
    ServoCAN_GoToAbsolutePosition(pos);
}
```

## 四、电机PWM控制 (motor_control.c)

```c
// TIM3 CH1/CH2: 两路PWM控制一个直流电机
// CH1: 正转PWM, CH2: 反转PWM
// 或者 CH1 PWM + CH2方向

void Motor_Control_Update(uint16_t pwm_value) {
    if (pwm_value == 0) {
        // 停止: 两路都低
        __HAL_TIM_SET_COMPARE(&htim3, TIM_CHANNEL_1, 0);
        __HAL_TIM_SET_COMPARE(&htim3, TIM_CHANNEL_2, 0);
    } else if (current_direction == DIR_FORWARD) {
        // 前进
        __HAL_TIM_SET_COMPARE(&htim3, TIM_CHANNEL_1, pwm_value);
        __HAL_TIM_SET_COMPARE(&htim3, TIM_CHANNEL_2, 0);
    } else {
        // 后退
        __HAL_TIM_SET_COMPARE(&htim3, TIM_CHANNEL_1, 0);
        __HAL_TIM_SET_COMPARE(&htim3, TIM_CHANNEL_2, pwm_value);
    }
}
```

## 五、轮速测量 (wheel_speed.c)

```c
// TIM4 CH3/CH4: 霍尔传感器脉冲计数
// 每个脉冲 = 电机旋转一圈的一部分

void WheelSpeed_CaptureCallback(TIM_HandleTypeDef* htim) {
    if (htim->Instance == TIM4) {
        // 上升沿触发捕获, 计数脉冲
        s_pulse_count_left++;
        // 计算周期: 用于瞬时速度
        uint32_t period = __HAL_TIM_GET_CAPTURE(&htim4, TIM_CHANNEL_3);
        // 速度 = 常数 / 周期
    }
}

void WheelSpeed_Task(void) {
    static uint32_t last_calc = 0;
    uint32_t now = HAL_GetTick();
    
    if (now - last_calc >= 100) {  // 每100ms计算一次
        // 100ms内的脉冲数 → 速度
        float speed = (float)s_pulse_count_left / PULSE_PER_REV 
                     * WHEEL_CIRCUMFERENCE * 3600; // km/h
        s_current_speed_01kmh = (int16_t)(speed * 10);
        
        // 清零计数器
        s_pulse_count_left = 0;
        s_pulse_count_right = 0;
        last_calc = now;
    }
}
```

## 六、串口协议解析 (train_uart.c)

```c
#define SOF_BYTE0 0xAA
#define SOF_BYTE1 0x55
#define FRAME_LEN 22

uint8_t train_uart_rx_buf[256];
uint8_t train_uart_rx_head = 0, train_uart_rx_tail = 0;

// 中断: 逐字节存入环形缓冲区
void USART1_IRQHandler(void) {
    if (USART1->SR & USART_SR_RXNE) {
        uint8_t byte = USART1->DR;
        train_uart_rx_buf[train_uart_rx_tail] = byte;
        train_uart_rx_tail = (train_uart_rx_tail + 1) % 256;
    }
}

// 主循环: 提取帧
uint8_t TrainUART_PollFollowCmd(FollowCmd_t* cmd) {
    while (head != tail) {
        // 查找SOF
        uint8_t b0 = train_uart_rx_buf[head];
        if (b0 == SOF_BYTE0) {
            uint8_t b1 = train_uart_rx_buf[(head+1) % 256];
            if (b1 == SOF_BYTE1) {
                // 读取整帧
                uint8_t frame[FRAME_LEN];
                for (int i = 0; i < FRAME_LEN; i++) {
                    frame[i] = train_uart_rx_buf[(head+i) % 256];
                }
                // CRC校验
                uint16_t crc_calc = CRC16_MODBUS(frame, 20);
                uint16_t crc_recv = (frame[21] << 8) | frame[20];
                if (crc_calc == crc_recv) {
                    // 解析TYPE=0x10的跟随指令
                    if (frame[3] == 0x10) {
                        cmd->steer_01 = (frame[13] << 8) | frame[12];
                        cmd->thr_ref = (frame[15] << 8) | frame[14];
                        cmd->brake = frame[16];
                        head = (head + FRAME_LEN) % 256;
                        return 1;
                    }
                }
                // CRC错误: 跳过SOF继续搜索
            }
        }
        head = (head + 1) % 256;
    }
    return 0;
}
```
